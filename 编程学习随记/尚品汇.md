//配置src文件夹的别名，为@
//vite.config.ts
import { defineConfig } from 'vite'
import vue from '@vitejs/plugin-vue'
import path from "path";

// https://vitejs.dev/config/
export default defineConfig({
  plugins: [vue()],
  resolve: {
    alias: {
      '@': path.resolve(__dirname, '/src'),
    },
  }

})

tsconfig.json
    "baseUrl": ".",
    "paths": {
      "@/*": [
        "src/*"
      ]
  }
